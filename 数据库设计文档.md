### 2.2 表结构设计

#### 用户表 (users)

```sql
CREATE TABLE users (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  username VARCHAR(50) UNIQUE NOT NULL,
  nickname VARCHAR(100),
  avatar_url VARCHAR(255),
  level VARCHAR(20) DEFAULT 'beginner',
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_users_username ON users(username);
```

#### 分析记录表 (analysis_records)

```sql
CREATE TABLE analysis_records (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id INTEGER,
  session_id VARCHAR(64),
  action_type VARCHAR(20) NOT NULL,
  action_name VARCHAR(50),
  angle VARCHAR(10),
  score INTEGER CHECK(score >= 0 AND score <= 100),
  level VARCHAR(20),
  duration_ms INTEGER,
  keypoints_json TEXT,
  differences_json TEXT,
  file_path VARCHAR(255),
  file_type VARCHAR(10),
  file_size INTEGER,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE INDEX idx_records_user ON analysis_records(user_id);
CREATE INDEX idx_records_action ON analysis_records(action_type);
CREATE INDEX idx_records_created ON analysis_records(created_at);
```

#### 标准动作表 (standard_actions)

```sql
CREATE TABLE standard_actions (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  action_code VARCHAR(30) UNIQUE NOT NULL,
  action_name VARCHAR(50) NOT NULL,
  category VARCHAR(20) NOT NULL,
  level VARCHAR(20) NOT NULL,
  player_name VARCHAR(50),
  description TEXT,
  video_side_path VARCHAR(255),
  video_front_path VARCHAR(255),
  keypoints_json TEXT NOT NULL,
  common_errors_json TEXT,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_actions_category ON standard_actions(category);
CREATE INDEX idx_actions_level ON standard_actions(level);
```

#### 训练计划表 (training_plans)

```sql
CREATE TABLE training_plans (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id INTEGER NOT NULL,
  plan_name VARCHAR(100) NOT NULL,
  target_actions_json TEXT,
  start_date DATE,
  end_date DATE,
  status VARCHAR(20) DEFAULT 'active',
  progress INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE INDEX idx_plans_user ON training_plans(user_id);
CREATE INDEX idx_plans_status ON training_plans(status);
```

#### 训练记录表 (training_sessions)

```sql
CREATE TABLE training_sessions (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  plan_id INTEGER,
  user_id INTEGER NOT NULL,
  action_id INTEGER NOT NULL,
  duration_seconds INTEGER,
  rep_count INTEGER,
  avg_score INTEGER,
  best_score INTEGER,
  notes TEXT,
  session_date DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (plan_id) REFERENCES training_plans(id),
  FOREIGN KEY (user_id) REFERENCES users(id),
  FOREIGN KEY (action_id) REFERENCES standard_actions(id)
);

CREATE INDEX idx_sessions_plan ON training_sessions(plan_id);
CREATE INDEX idx_sessions_user ON training_sessions(user_id);
CREATE INDEX idx_sessions_date ON training_sessions(session_date);
```

## 五、数据迁移方案

### 5.1 从 JSON 迁移到 SQLite

```javascript
// migrate.js
import sqlite3 from 'sqlite3'
import fs from 'fs'

const db = new sqlite3.Database('./data/pingpong.db')
const jsonData = JSON.parse(fs.readFileSync('./data/history.json', 'utf8'))

db.serialize(() => {
  jsonData.records.forEach(record => {
    db.run(`
      INSERT INTO analysis_records 
      (action_type, angle, score, level, file_path, created_at)
      VALUES (?, ?, ?, ?, ?, ?)
    `, [
      record.actionType,
      record.angle,
      record.score,
      record.level,
      record.filePath,
      record.timestamp
    ])
  })
})
```
